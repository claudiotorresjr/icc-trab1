129 void preCondicionador(double p, double *M, double *A, long int tam){
130  long int i, j;
131  double *L = (double*)malloc(tam*tam*sizeof(double)); //matriz lower
132  double *U = (double*)malloc(tam*tam*sizeof(double)); //matriz upper
133  double *D = (double*)malloc(tam*tam*sizeof(double)); //matriz diagonal
134  double *R1 = (double*)malloc(tam*tam*sizeof(double)); //matriz resposta1
135  double *R2 = (double*)malloc(tam*tam*sizeof(double)); //matriz resposta2
136 
137  if(p == 0.0){//sem pre-condicionador
138  for(i = 0; i < tam; i++){
139  for(j = 0; j < tam; j++){
140  if(i == j){
141  M[i*tam + j] = 1.0;
142  }else{
143  M[i*tam + j] = 0.0;
144  }
145  }
146  }
147  }else if(p > 0.0 && p < 1.0){//pre-condicionador de jacobi
148  for(i = 0; i < tam; i++){
149  for(j = 0; j < tam; j++){
150  if(i == j){
151  M[i*tam + j] = A[i*tam + j];
152  }else{
153  M[i*tam + j] = 0.0;
154  }
155  }
156  }
157  }else{//pré-condicionador de Gauss-Seidel p/ x = 1 e pré-condicionador SSOR p/ 1 < x < 2
158 
159  criaMatrizes(A, L, U, D, tam);
160  for(i = 0; i < tam; i++){
161  for(j = 0; j < tam; j++){
162  R1[i*tam + j] = D[i*tam + j] + p*L[i*tam + j];
163  }
164  }
165  for(i = 0; i < tam; i++){
166  for(j = 0; j < tam; j++){
167  R2[i*tam + j] = D[i*tam + j] + p*U[i*tam + j];
168  }
169  }
170  for(i = 0; i < tam; i++){
171  for(j = 0; j < tam; j++){
172  if(i == j){
173  D[i*tam + j] = 1 / D[i*tam + j];
174  }
175  }
176  }
177  multMatMat(R1, D, tam, U); //matriz U sendo usada como intermediaria da operação
178  multMatMat(U, R2, tam, M);
179  Cholesky(M, tam);
180  }
181  free(L);
182  free(U);
183  free(D);
184  free(R1);
185  free(R2);

247 void Cholesky(double *M, long int tam)
248 {
249  long int i, j, k;
250 
251  for(k = 0; k < tam; k++){
252  M[k*tam + k] = sqrtf(M[k*tam + k]);
253  for(i = k + 1; i < tam; i++){
254  if(M[i*tam + k] != 0.0){
255  M[i*tam + k] = M[i*tam + k] / M[k*tam + k];
256  }
257  }
258  for(j = k + 1; j < tam; j++){
259  for(i = j; i < tam; i++){
260  if(M[i*tam + j] != 0.0){
261  M[i*tam + j] = M[i*tam + j] - M[i*tam + k]*M[j*tam + k];
262  }
263  }
264  }
265  }
266  for(i = 0; i < tam; i++){
267  for(j = i + 1; j < tam; j++){
268  M[i*tam + j] = 0.0;
